☁️ Google 試算表 雲端同步教學

完全免費！透過 Google 試算表，您可以讓兩台平板（老闆與客人）即時同步商店設定與庫存。

步驟 1：建立試算表並貼上程式碼

  1. 前往您的 Google 雲端硬碟，新增一個全新的空白「Google 試算表」。
  2. 點擊上方選單的 「擴充功能」 ➔ 「Apps Script」。
  3. 將裡面的英文全部刪除，並貼上剛剛複製的程式碼。
function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var doc = SpreadsheetApp.getActiveSpreadsheet(); var sheet = doc.getSheets()[0]; if (sheet.getLastRow() === 0) { sheet.appendRow(["最後更新時間", "資料專屬ID", "完整內容(JSON)"]); } var payload = JSON.parse(e.postData.contents); var timestamp = new Date().toLocaleString('zh-TW', { timeZone: 'Asia/Taipei' }); var dataRange = sheet.getDataRange(); var values = dataRange.getValues(); var rowIndex = -1; for (var i = 1; i < values.length; i++) { if (values[i][1] === payload.id) { rowIndex = i + 1; break; } } var jsonStr = typeof payload.data === 'string' ? payload.data : JSON.stringify(payload.data); if (rowIndex !== -1) { sheet.getRange(rowIndex, 1, 1, 3).setValues([[timestamp, payload.id, jsonStr]]); } else { sheet.appendRow([timestamp, payload.id, jsonStr]); } return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON); } catch(ex) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": ex.message})).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } function doGet(e) { var doc = SpreadsheetApp.getActiveSpreadsheet(); var sheet = doc.getSheets()[0]; var result = { data: null }; if (sheet.getLastRow() === 0) return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); var values = sheet.getDataRange().getValues(); for (var i = 1; i < values.length; i++) { var id = values[i][1]; var jsonStr = values[i][2]; if (id === 'shopping_data' && jsonStr) { try { result.data = JSON.parse(jsonStr); } catch(err){} break; } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); }

步驟 2：部署為網頁應用程式

  1. 點擊右上角藍色按鈕 「部署」 ➔ 「新增部署作業」。
  2. 點選左邊齒輪，選擇 「網頁應用程式」。
  3. 【非常重要】「誰可以存取」請務必下拉選擇 「所有人 (Anyone)」。
  4. 點擊部署並核准權限後，您會得到一串「網頁應用程式網址」，請複製它。

步驟 3：在此貼上網址並儲存

請輸入有效的 Google Apps Script 網址 (須以 http 開頭)。
